        const { data: { user }, error } = await supabase.auth.getUser();
        if (error || !user) return null;

        const { data: profile } = await supabase.from('profiles').select('role').eq('id', user.id).single();

    async function getCurrentCompany() {
        // For owner/staff: get the first company they belong to (simplified V1)
        const { data: membership } = await supabase.from('company_members').select('company_id').limit(1).single();

        return null;

        async function loadDashboard() {

            const { data: incomes } = await supabase.from('transactions').select('amount').eq('company_id', companyId).eq('type', 'income').gte('transaction_date', new Date(new Date().getFullYear(), new Date().getMonth(), 1).toISOString().split('T')[0]);

            const { data: expenses } = await supabase.from('transactions').select('amount').eq('company_id', companyId).eq('type', 'expense').gte('transaction_date', new Date(new Date().getFullYear(), new Date().getMonth(), 1).toISOString().split('T')[0]);

            document.getElementById('tabContent').innerHTML = `
                <div class="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-4 gap-4 mb-6">
                    <div class="bg-white p-4 rounded shadow">Pendapatan Bulan Ini<br><span class="text-2xl font-bold text-green-600">RM ${totalIncome.toFixed(2)}</span></div>

                    <div class="bg-white p-4 rounded shadow">Baki Semasa<br><span class="text-2xl font-bold">RM ${(totalIncome - totalExpense).toFixed(2)}</span></div>
                </div>
                <div class="flex gap-2 flex-wrap">
                    <button id="addIncomeBtn" class="bg-green-600 text-white px-4 py-2 rounded">Tambah Pendapatan</button>

                </div>
                <div id="txnForm" class="mt-4 hidden bg-white p-4 rounded"></div>
            `;
            document.getElementById('addIncomeBtn').onclick = () => showTransactionForm(companyId, 'income');
            document.getElementById('addExpenseBtn').onclick = () => showTransactionForm(companyId, 'expense');

        async function showTransactionForm(companyId, type) {
            const formDiv = document.getElementById('txnForm');
            formDiv.innerHTML = `
                <h3 class="font-bold mb-2">Tambah ${type === 'income' ? 'Pendapatan' : 'Perbelanjaan'}</h3>
                <input id="amount" type="number" placeholder="Jumlah" class="border p-2 w-full mb-2">
                <input id="desc" placeholder="Keterangan" class="border p-2 w-full mb-2">
                <input id="date" type="date" class="border p-2 w-full mb-2">
                <button id="saveTxn" class="bg-blue-600 text-white px-4 py-2 rounded">Simpan</button>
            `;
            formDiv.classList.remove('hidden');
            document.getElementById('saveTxn').onclick = async () => {
                const amount = parseFloat(document.getElementById('amount').value);
                const description = document.getElementById('desc').value;
                const date = document.getElementById('date').value;
                if (!amount) return;
                const { error } = await supabase.from('transactions').insert({
                    company_id: companyId,
                    user_id: (await getCurrentUser()).user.id,
                    type: type,
                    amount: amount,
                    description: description,
                    transaction_date: date || new Date().toISOString().split('T')[0]
                });
                if (error) alert(error.message);
                else { alert('Disimpan'); loadDashboard(); }
            };
        }
        
        // Transactions tab
        async function loadTransactions() {
            const companyId = await getOwnerCompanyId();
            if (!companyId) { document.getElementById('tabContent').innerHTML = '<p>Tiada syarikat.</p>'; return; }
            const { data: txns } = await supabase.from('transactions').select('*').eq('company_id', companyId).order('created_at', { ascending: false });
            let html = `<div class="bg-white p-4 rounded shadow"><h3 class="font-semibold mb-2">Senarai Transaksi</h3><ul>`;
            txns?.forEach(t => {
                html += `<li class="border-b py-2 flex justify-between"><span>${t.description || t.type} - RM ${t.amount}</span><span class="text-sm text-gray-500">${t.transaction_date}</span></li>`;
            });
            html += `</ul><button id="addTxnBtn" class="mt-4 bg-blue-600 text-white px-4 py-2 rounded">Tambah Transaksi</button></div>`;
            document.getElementById('tabContent').innerHTML = html;
            document.getElementById('addTxnBtn').onclick = () => {
                // simple prompt for quick add
                const type = prompt('Jenis (income/expense)');
                const amount = prompt('Jumlah');
                if (amount) supabase.from('transactions').insert({ company_id: companyId, user_id: (await getCurrentUser()).user.id, type, amount: parseFloat(amount), transaction_date: new Date().toISOString().split('T')[0] }).then(() => loadTransactions());
            };
        }
        
        // Documents tab (upload to storage)
        async function loadDocuments() {
            const companyId = await getOwnerCompanyId();
            document.getElementById('tabContent').innerHTML = `
                <div class="bg-white p-4 rounded shadow">
                    <h3 class="font-semibold mb-2">Muat Naik Dokumen</h3>
                    <input type="file" id="fileInput" class="mb-2">
                    <select id="docType" class="border p-2 mb-2 w-full">
                        <option value="receipt">Resit</option>
                        <option value="invoice">Invois</option>
                        <option value="bank_statement">Penyata Bank</option>
                    </select>
                    <button id="uploadBtn" class="bg-blue-600 text-white px-4 py-2 rounded">Upload</button>
                    <div id="docList" class="mt-4"></div>
                </div>
            `;
            document.getElementById('uploadBtn').onclick = async () => {
                const file = document.getElementById('fileInput').files[0];
                if (!file) return;
                const fileType = document.getElementById('docType').value;
                const filePath = `${companyId}/${Date.now()}_${file.name}`;
                const { data, error } = await supabase.storage.from('documents').upload(filePath, file);
                if (error) { alert(error.message); return; }
                const { data: { publicUrl } } = supabase.storage.from('documents').getPublicUrl(filePath);
                await supabase.from('documents').insert({
                    company_id: companyId,
                    user_id: (await getCurrentUser()).user.id,
                    file_name: file.name,
                    file_url: publicUrl,
                    file_type: fileType,
                    storage_path: filePath
                });
                alert('Muat naik berjaya');
                loadDocuments();
            };
            // list documents
            const { data: docs } = await supabase.from('documents').select('*').eq('company_id', companyId);
            const listDiv = document.getElementById('docList');
            listDiv.innerHTML = '<h4 class="font-semibold">Dokumen sebelum ini:</h4><ul>';
            docs?.forEach(d => { listDiv.innerHTML += `<li>${d.file_name} (${d.file_type})</li>`; });
            listDiv.innerHTML += '</ul>';
        }

                const tabName = btn.getAttribute('data-tab');
                tabs[tabName]();
            };
        });
        // load default dashboard
        loadDashboard();
